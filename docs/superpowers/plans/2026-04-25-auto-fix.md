# Auto-Fix Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Enable Claude to automatically fix violations detected by CHP laws when configured.

**Architecture:** When a law fails verification, the fix-trigger checks the law's `autoFix` mode. If not "never", it invokes Claude with the law's guidance and violation context. Claude proposes a fix, shows the diff, and applies it on user confirmation (or automatically for "auto" mode).

**Tech Stack:** Bash (core runtime), jq (JSON processing), existing CHP infrastructure

# Validate the autoFix field if present
# Usage: validate_autofix_field <law_json_path>
# Returns: 0 if valid, 1 if invalid
validate_autofix_field() {
    local law_json="$1"
    local autofix_val
    autofix_val=$(jq -r '.autoFix // "never"' "$law_json" 2>/dev/null)

    if [[ -n "$autofix_val" && "$autofix_val" != "null" ]]; then
        local autofix_ok=false
        for v in $_CHP_VALID_AUTOFIX; do
            [[ "$autofix_val" == "$v" ]] && autofix_ok=true
        done
        if ! $autofix_ok; then
            echo "autoFix must be one of: $_CHP_VALID_AUTOFIX (got: '$autofix_val')" >&2
            return 1
        fi
    fi
    return 0
}

Create `core/fix-trigger.sh` with:

```bash
#!/usr/bin/env bash
# Fix Trigger - Invokes Claude to fix violations after verification failure

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
source "$SCRIPT_DIR/common.sh"
source "$SCRIPT_DIR/logger.sh"

logger_init

# Usage: trigger_fix <law_name> <hook_type> [hook_args...]
trigger_fix() {
    local law_name="$1"
    local hook_type="$2"
    shift 2

    local law_dir="$LAWS_DIR/$law_name"
    local law_json="$law_dir/law.json"
    local guidance_md="$law_dir/guidance.md"

    if [[ ! -f "$law_json" ]]; then
        log_error "Law not found: $law_name"
        return 1
    fi

    # Check autoFix mode
    local auto_fix
    auto_fix=$(jq -r '.autoFix // "never"' "$law_json" 2>/dev/null)

    if [[ "$auto_fix" == "never" ]]; then
        log_debug "Law '$law_name' has autoFix: never, skipping fix flow"
        return 0
    fi

    if [[ ! -f "$guidance_md" ]]; then
        log_warn "No guidance.md found for law: $law_name, cannot auto-fix"
        return 0
    fi

    log_info "Auto-fix available for law '$law_name' (mode: $auto_fix)"

    # Export context for the agent
    export CHP_FIX_LAW_NAME="$law_name"
    export CHP_FIX_MODE="$auto_fix"
    export CHP_FIX_HOOK_TYPE="$hook_type"
    export CHP_FIX_LAW_DIR="$law_dir"
    export CHP_FIX_GUIDANCE="$guidance_md"

    # Get affected files for context
    local affected_files=""
    case "$hook_type" in
        pre-commit)
            affected_files=$(git diff --cached --name-only 2>/dev/null | tr '\n' ' ')
            ;;
        pre-push|post-commit)
            if git rev-parse @{u} >/dev/null 2>&1; then
                affected_files=$(git diff --name-only HEAD @{u} 2>/dev/null | tr '\n' ' ')
            else
                affected_files=$(git diff --name-only HEAD^..HEAD 2>/dev/null | tr '\n' ' ')
            fi
            ;;
    esac
    export CHP_FIX_FILES="$affected_files"

    # Look for fixer agent prompt
    local fixer_prompt="$CHP_BASE/agents/fixer.md"
    if [[ ! -f "$fixer_prompt" ]]; then
        log_warn "Fixer agent prompt not found at $fixer_prompt"
        return 0
    fi

    # Read and output the fixer prompt for Claude to see
    cat "$fixer_prompt"
    logger_log "fix_trigger" "law_name" "$law_name" "mode" "$auto_fix" "hook_type" "$hook_type"

    return 0
}

# If run directly (not sourced)
if [ "${BASH_SOURCE[0]}" = "$0" ]; then
    if [[ $# -lt 2 ]]; then
        log_error "Usage: $0 <law_name> <hook_type> [hook_args...]"
        exit 1
    fi

    trigger_fix "$@"
    exit $?
fi
```

## Task 6: Create chp-fix CLI Command

**Files:**
- Create: `commands/chp-fix`

- [ ] **Step 1: Create the chp-fix command**

Create `commands/chp-fix` with:

```bash
#!/usr/bin/env bash
# CHP Fix - Manually trigger auto-fix for violations

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
source "$SCRIPT_DIR/../core/common.sh"
source "$SCRIPT_DIR/../core/logger.sh"
source "$SCRIPT_DIR/../core/fix-trigger.sh"

logger_init

show_help() {
    cat << EOF
CHP Fix - Manually trigger auto-fix for law violations

Usage:
  chp fix                    # List fixable violations and offer to fix
  chp-fix --apply           # Auto-apply all "auto" mode fixes
  chp-fix --law <name>      # Fix specific law
  chp-fix --dry-run         # Show what would be fixed
  chp-fix --help            # Show this help

Options:
  --apply       Apply fixes without confirmation (for "auto" mode laws)
  --law <name>  Only attempt fixes for the specified law
  --dry-run     Show what would be fixed without applying
  --help        Show this help message
EOF
}

list_fixable_laws() {
    echo "Fixable Laws:"
    echo "=============="
    for law_dir in "$LAWS_DIR"/*; do
        [[ ! -d "$law_dir" ]] && continue
        local law_name=$(basename "$law_dir")
        local law_json="$law_dir/law.json"
        [[ ! -f "$law_json" ]] && continue

        local auto_fix
        auto_fix=$(jq -r '.autoFix // "never"' "$law_json" 2>/dev/null)
        [[ "$auto_fix" == "never" ]] && continue

        local failures
        failures=$(jq -r '.failures // 0' "$law_json" 2>/dev/null)
        local severity
        severity=$(jq -r '.severity // "warn"' "$law_json" 2>/dev/null)

        echo "  $law_name (mode: $auto_fix, severity: $severity, failures: $failures)"
    done
}

# Parse arguments
LAW_NAME=""
APPLY_AUTO=false
DRY_RUN=false

while [[ $# -gt 0 ]]; do
    case "$1" in
        --help|-h)
            show_help
            exit 0
            ;;
        --law)
            LAW_NAME="$2"
            shift 2
            ;;
        --apply)
            APPLY_AUTO=true
            shift
            ;;
        --dry-run)
            DRY_RUN=true
            shift
            ;;
        *)
            echo "Unknown option: $1"
            show_help
            exit 1
            ;;
    esac
done

if [[ "$DRY_RUN" == true ]]; then
    echo "Dry run mode - showing what would be fixed:"
    echo ""
    list_fixable_laws
    exit 0
fi

if [[ -n "$LAW_NAME" ]]; then
    # Fix specific law
    if [[ ! -d "$LAWS_DIR/$LAW_NAME" ]]; then
        log_error "Law not found: $LAW_NAME"
        exit 1
    fi

    log_info "Triggering fix for law: $LAW_NAME"
    trigger_fix "$LAW_NAME" "manual"
else
    # List fixable laws
    list_fixable_laws

    if [[ "$APPLY_AUTO" == true ]]; then
        echo ""
        log_info "Applying all 'auto' mode fixes..."
        for law_dir in "$LAWS_DIR"/*; do
            [[ ! -d "$law_dir" ]] && continue
            local law_name=$(basename "$law_dir")
            local law_json="$law_dir/law.json"
            [[ ! -f "$law_json" ]] && continue

            local auto_fix
            auto_fix=$(jq -r '.autoFix // "never"' "$law_json" 2>/dev/null)
            [[ "$auto_fix" != "auto" ]] && continue

            trigger_fix "$law_name" "manual"
        done
    fi
fi
```
